\section*{Part I — Algorithmic Overview}
\label{sec:overview}

The system tracks per-object Bernoulli-Gaussian beliefs over rigid
body poses on $\SE$, conditioned on RGB-D detections from a moving
camera mounted on a manipulator. Per frame $k$ it ingests the SLAM
base-pose $T_{wb,k}$, the camera-to-base extrinsic $T_{bc,k}$, the
gripper-to-base $T_{bg,k}$, the gripper finger width, and a list of
SAM2 detections $\{(\mathrm{pid}_d, \mathrm{label}_d, \mathrm{mask}_d,
\mathrm{score}_d)\}_d$ projected through depth.

\paragraph{1. State.} Each track $o$ carries
$(\mu_{bo}, \Sigma_{bo}, r_o, \ell_o, \tau_o)$:
$\mu_{bo} \in \SE$ is the object pose in the \emph{base} frame,
$\Sigma_{bo} \in \R^{6\times 6}$ is its covariance in the body-frame
tangent at $\mu_{bo}$, $r_o \in [0,1]$ is the existence probability,
$\ell_o$ a label histogram, and $\tau_o$ the SAM2 perception id.
\emph{Base-frame storage is the canonical choice}: SLAM uncertainty
$\Sigma_{wb}$ never enters the recursion; the world view is the
deterministic composition $T_{wo} = T_{wb}\,\mu_{bo}$.

\paragraph{2. Predict.} Two motion models, applied per-track:
world-static (default, $u_k = T_{wb,k}^{-1}T_{wb,k-1}$, applied via
the adjoint to $\Sigma$ plus per-phase Q) or rigid-attached
(held set, $\delta_{bg,k} = T_{bg,k}T_{bg,k-1}^{-1}$).

\paragraph{3. Detection conditioning.} Subpart suppression
(\file{det\_dedup.py}) absorbs detections whose 3D voxel sets are
mostly contained in another's; centroid back-projection drops masks
without valid depth.

\paragraph{4. Association.} Translation-only Mahalanobis with a
$\chi^2_3$ gate, $G_{\text{out},t}=21.108$, $R_{cam}=(0.02\,m)^2 I$.
Hungarian on the gated cost; SAM2-id continuity bonus $\alpha=4.4$
breaks ties.

\paragraph{5. Update.} For matched pairs on the held track, two
safety prefilters guard the Joseph step:
held-prefilter (centroid-to-$T_{we}$ within 0.25~m) and
innovation-clamp (centroid-to-$\mu_w$ within 0.20~m). The Joseph step
runs on the \emph{translation block only}; the rotation block and
cross-cov are left zeroed (a translation-only observation carries no
rotation information). This keeps $\Sigma_{bo}$ PSD across long runs.

\paragraph{6. Birth and prune.} A confirm-3 / score-0.20 / border /
near-live (default 5~cm; 25~cm against $T_{we}$ for the held oid)
chain controls births. Prune when $r<10^{-3}$, except hold the held
oid above $r=0.5$.

\paragraph{7. Self-merge.} Same-label tracks within 5~cm fuse via
information sum, EXCEPT pairs in the relation-protected set
(scene graph asserts they are distinct). The held seed is always
the keeper.

\paragraph{8. Relations.} A VLM/REST backend
(\file{relation\_orchestrator.py}) is fired at first call,
release transitions, new-oid arrivals, and every 90 frames. Its
parent matrix is converted to \texttt{RelationEdge} records with the
convention \textbf{parent = supported, child = supporter} (LLM's $i$
is the supporter; we emit $(j,i,\text{on})$). A per-edge EMA
($\alpha=0.3$, threshold $0.5$) smooths flicker.

\paragraph{9. Held-set expansion.} The set carried by the gripper is
the closure of $\{\text{held seed}\}$ under in/on edges; after each
self-merge, EMA keys are remapped (drop $\to$ keep) so the closure
stays valid.

\section*{Part II — Detailed Elaboration}
\label{sec:detail}

\subsection*{II.1 State}

\file{pose\_update/gaussian\_state.py}.
$$\mathcal{S}_o = (\mu_{bo}\in\SE,\ \Sigma_{bo}\in\R^{6\times 6}_{\succeq 0},\
  r_o\in[0,1],\ \ell_o,\ \tau_o\in\mathbb{Z},\ \text{age},\ \text{frames\_since\_obs}).$$

Body-frame tangent at $\mu_{bo}$: a perturbation $\xi\in\se$ acts as
$\mu' = \mu\,\Exp(\xi)$ (right-multiply). Ordering
$\xi=(\xi_v;\xi_\omega)$ — \emph{translation first then rotation} (matches
\file{pose\_update/ekf\_se3.py}, lines 5--6, 51, 108). The adjoint
$$\Ad(T) = \begin{bmatrix} R & [t]_\times R \\ 0 & R \end{bmatrix}$$
moves a tangent vector through $T=(R,t)$.

\subsection*{II.2 Predict}

\textbf{World-static.} A world-static object satisfies
$T_{wb,k}\mu_{bo,k} = T_{wb,k-1}\mu_{bo,k-1}$, hence
$\mu_{bo,k} = u_k\,\mu_{bo,k-1}$ with
$u_k = T_{wb,k}^{-1}T_{wb,k-1}$.

The covariance recursion now branches on whether the track was
observed last frame (\file{pose\_update/gaussian\_state.py}\,$\to$
\texttt{predict\_static}, with caller-supplied
\texttt{unobserved\_oids}):
\begin{equation}\label{eq:proc}
\Sigma_{bo,k} \;=\;
\begin{cases}
\Phi\!\bigl(\Ad(u_k)\,\Sigma_{bo,k-1}\,\Ad(u_k)^\top + Q_{\text{static}}\bigr) & \text{observed at } k{-}1, \\[2pt]
\Sigma_{bo,k-1} & \text{unobserved at } k{-}1 \text{ (frozen keyframe)}.
\end{cases}
\end{equation}
The mean update $\mu_{bo,k}=u_k\mu_{bo,k-1}$ runs in \emph{both}
branches, deterministically. $Q_{\text{static}}$ schedules on
\texttt{frames\_since\_obs}: $1\!{\rm e}{-6}I_6$ (idle, $<5$),
$1\!{\rm e}{-5}I_6$ ($<50$), $1\!{\rm e}{-8}I_6$ thereafter
(\file{ekf\_se3.py}\,$\to$ \texttt{process\_noise\_for\_phase}).
$\Phi$ is the optional covariance-saturation operator
(\file{ekf\_se3.py}\,$\to$ \texttt{saturate\_covariance})
\begin{equation}\label{eq:phi}
\Phi(P) \;=\; \min\!\bigl(1,\;\Tr P_{\max}/\Tr P\bigr)\,P,
\end{equation}
inactive when \texttt{cfg.P\_max} is \texttt{None} (the default).
Bernoulli predict on existence is
\begin{equation}\label{eq:bern_pred_r}
r_{k|k-1} = p_s\,r_{k-1|k-1}, \qquad p_s = 1 \text{ default}
\end{equation}
(\file{bernoulli.py}\,$\to$\,\texttt{r\_predict}).

\paragraph{Why the unobserved branch freezes $\Sigma_{bo}$ (keyframe
mechanism).}
The Ad-conjugate transport carries a cross-coupling
$[t(u_k)]_\times P_{rr}[t(u_k)]_\times^\top$ that pumps rotation
uncertainty into the translation block each frame. For tracks with
high $\sigma_{yaw}$ (e.g.\ a freshly-released bottle at
$\sigma_{yaw}=\pi$ from \eqref{eq:gravity_sigma}), the per-frame
contribution $\Delta x^2\!\cdot\!\pi^2$ compounds catastrophically
($\sim 1.2\,$m over 60 frames at apple\_drop). The keyframe-mechanism
fix exploits the algebraic identity that, telescoping the
deterministic mean update over $K$ frames since the last observation
at $k_f$,
\begin{equation}\label{eq:telescope}
\mu_{bo,K} \;=\; u_K \cdots u_{k_f+1}\,\mu_{bo,k_f}
            \;=\; T_{wb,K}^{-1}\,T_{wb,k_f}\,\mu_{bo,k_f}.
\end{equation}
Composing with the current SLAM pose then gives
$T_{wo,K}=T_{wb,K}\mu_{bo,K}=T_{wb,k_f}\mu_{bo,k_f}$ — \emph{constant}
for a static object, exactly as required. The keyframe is not stored
explicitly; $\Sigma_{bo,K}=\Sigma_{bo,k_f}$ is preserved by skipping
the cov update. World composition (\eqref{eq:world_compose} in §II.5)
then uses the \emph{current} $\Sigma_{wb,K}$ for the lever-arm
contribution, so the world-frame uncertainty grows only via the SLAM
marginal — never via per-frame predict drift. Verified to machine
precision on apple\_drop oid=1 (fso=224 frames):
$\|\mu_{bo,K} - T_{wb,K}^{-1}T_{wb,k_f}\mu_{bo,k_f}\|=2.22\!\cdot\!10^{-14}$.

\textbf{Rigid-attached.} For tracks in the held set $\mathcal{H}_k$,
$\mu_{bo,k} = \delta_{bg,k}\,\mu_{bo,k-1}$ with
$\delta_{bg,k} = T_{bg,k}T_{bg,k-1}^{-1}$. The driver applies a
phase-aware $Q_{\text{manip}}$:
\begin{itemize}
\item Seed during grasping/releasing: $\mathrm{diag}((0.02)^2{\times}3,\,(0.10)^2{\times}3)$ — physical grip-slack.
\item Seed during holding: $\mathrm{diag}(2.5\!{\rm e}{-4}{\times}3,\,1\!{\rm e}{-3}{\times}3)$ — settled grip.
\item Transitive members (apples on tray): always the holding row — they don't deform when the gripper closes around the seed.
\end{itemize}
The held branch still runs the full Ad-conjugate (no
keyframe-mechanism shortcut), so the same translation/rotation
cross-pumping applies — bounded in practice by
$\|t(\delta_{bg,k})\|\!\ll\!1$ but \emph{not} eliminated. Edge cases
with $\sigma_{yaw}\!\to\!\pi$ inside the holding window
(re-grasps, release-transition) remain a known caveat; cap-rotation
mitigations are an open item.

\subsection*{II.3 Detection conditioning}

\textbf{Subpart suppression} (\file{det\_dedup.py}).
Voxelise each mask through depth at 2~cm; for ordered pair $(a,b)$
with $|V_b|\le|V_a|$, if $|V_b\cap V_a|/|V_b|>0.8$, absorb $b$ into $a$
(label histograms merge). Catches SAM2 mask splits / re-prompts.

\textbf{Centroid validity}.
$c_d^{\text{cam}} = \frac{1}{|m_d\cap\text{valid-depth}|}\sum_{p\in m_d}\mathrm{back\_project}(p,\mathrm{depth},K)$.
If the mask has no valid-depth pixels, $d$ is dropped (logged in
\texttt{centroid\_dropped}).

\subsection*{II.4 Association}

\file{pose\_update/association.py}\,$\to$\,\texttt{hungarian\_associate}.
For each $(o,d)$:
\begin{align*}
  t_{bo}^{\text{meas}} &= T_{bc}\bigl[c_d^{\text{cam}}; 1\bigr]_{1:3} \\
  R_{bo}^t &= R_{bc}\,R_{cam}\,R_{bc}^\top, \quad R_{cam} = \mathrm{diag}((0.02\,m)^2{\times}3) \\
  \nu_{od} &= t_{bo}^{\text{meas}} - \mu_{bo}[1{:}3, 4],\quad
  S_{od} = P_{tt,o} + R_{bo}^t \\
  d^2_{t,od} &= \nu_{od}^\top S_{od}^{-1} \nu_{od}.
\end{align*}

\paragraph{Cost.}
Two modes, selected by \texttt{BernoulliConfig.enforce\_label\_match}:
\begin{equation}\label{eq:cost_sam2}
C_{od} \;=\; d^2_{t,od}
  \;-\;\alpha_{\text{SAM2}}\,\bm 1[\tau_o = \mathrm{pid}_d,\ \tau_o\ge 0]
  \;+\;\beta\,\bm 1[\hat\ell_d \notin \mathcal H_o]
  \;+\;\gamma\,\max(0,\,1-\hat s_d).
\end{equation}
\textbf{Hard label gate} (\texttt{enforce\_label\_match=True}, paper
baseline): $\beta = \gamma = 0$, and any cell with mismatched label is
forced to the sentinel $\_\text{INFEASIBLE}=10^{12}$ (big-M).
\textbf{Soft mode} (\texttt{enforce\_label\_match=False}, used by
\file{tests/visualize\_ekf\_tracking.py}): $\beta=$
\texttt{hungarian\_label\_penalty} (default 0.0; visualizer sets 6.0)
and $\gamma=$ \texttt{hungarian\_score\_weight} (default 0.0; visualizer
sets 2.0). $\alpha_{\text{SAM2}}=4.4$ in either mode (continuity bonus).

\paragraph{Label-purity gate} (soft mode only).
The membership test $\hat\ell_d \in \mathcal H_o$ is not pure dict
membership: it is the helper
\texttt{\_label\_in\_history\_meaningful(t\_history, d\_label, min\_obs=5,
min\_share=0.05)} in \file{association.py}. A label is considered
``meaningfully in the history'' iff it has $\ge 5$ observations \emph{and}
$\ge 5\%$ of the total. This rejects stale trace-amount labels (e.g.\
3 obs against 53\,990 → \texttt{False}) that would otherwise bypass the
$\beta$ penalty.

\paragraph{Outer gate.}
The Hungarian solver minimises $\sum C_{o\sigma(o)}$. Two gates apply:
\begin{enumerate}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item Probabilistic chi$^2$: $d^2_{t,od} > G_{\text{out},t}=21.108$
  ($\chi^2_3$ at $99.97\%$) → cell infeasible.
\item Hard Euclidean residual cap: $\|\nu_{od}\|_2 > $
  \texttt{cfg.max\_residual\_m} → cell infeasible. Default $0.30$\,m;
  set \texttt{None} to disable. Catches the inflated-covariance
  pathology where, after a long miss run with $\sigma_t \approx 0.4$\,m,
  a $0.86$\,m residual gives $d^2_t \approx 17 < 21.108$ and the chi$^2$
  gate becomes meaningless.
\end{enumerate}
\begin{equation}\label{eq:hung}
\sigma^* = \arg\min_\sigma \sum_{o} C_{o\,\sigma(o)},\qquad
\text{drop } (o,d) \text{ if } C_{od} \ge \_\text{INFEASIBLE}.
\end{equation}

\subsection*{II.5 Update}

\textbf{Held-track prefilters} (\file{visualize\_ekf\_tracking.py},
matched-update branch). For each accepted match $(o,d)$ with $o$ the
held seed:
\begin{itemize}
\item Held-prefilter: $\|c_d^w - (T_{wb}T_{bg})_{1:3,4}\|_2 \le 0.25$~m.
\item Innovation clamp: $\|c_d^w - (T_{wb}\mu_{bo})_{1:3,4}\|_2 \le 0.20$~m.
\end{itemize}
A failure marks the detection consumed (so it doesn't re-birth) and
skips the EKF step; the rigid-attach predict carries the track.

\textbf{Joseph step (translation-only)}, \file{gaussian\_state.py}\,$\to$
\texttt{update\_observation\_centroid}. Let $K_{tt} = P_{tt}\,S^{-1}$ ($3\times 3$):
\begin{align*}
  \delta_t^{\text{base}} &= K_{tt}\,\nu \\
  \mu_{bo}[1{:}3, 4] &\leftarrow \mu_{bo}[1{:}3, 4] + \delta_t^{\text{base}} \\
  P_{tt}^{\text{post}} &= (I - K_{tt})\,P_{tt}\,(I - K_{tt})^\top + K_{tt}\,R_{bo}^t\,K_{tt}^\top \\
  \Sigma_{bo}^{\text{post}} &= \mathrm{block\!-\!diag}\bigl(P_{tt}^{\text{post}},\ P_{rr}\bigr).
\end{align*}
The cross-covariance is zeroed: a translation-only observation
provides no information about rotation (we set $R_{rot}=\infty$),
so the honest posterior is block-diagonal. An earlier version
preserved the prior cross-cov, which broke
$P_{rr}-P_{rt}^\top P_{tt}^{-1}P_{rt}\succeq 0$ over many updates;
the refactored version is PSD by construction, with an
\texttt{eigh}-clip belt-and-suspenders.

For pairs where ICP succeeded (\file{icp\_pose.py}), an alternative
full-$\SE$ update via \texttt{joseph\_update} in
\file{object\_belief.py} replaces the centroid path. The lift to base
frame is
\begin{equation}\label{eq:meas_compose}
T_{bo}^{\text{meas}} = T_{bc}\,\hat T^{co}, \qquad
R^{bo} = \Ad(T_{bc})\,\hat R^{co}\,\Ad(T_{bc})^\top,
\end{equation}
followed by IEKF inner iterations and the Joseph form
(\file{ekf\_se3.py}\,$\to$\,\texttt{ekf\_update};
\file{object\_belief.py}\,$\to$\,\texttt{joseph\_update};
\file{ekf\_se3.py}\,$\to$\,\texttt{huber\_weight} for the
inner soft / outer hard reweighting on $d^2_t$, \eqref{eq:huber_w}). The
EKF likelihood used by \eqref{eq:r_assoc} is
\begin{equation}\label{eq:ekf_lik}
\log\mathcal L = -\tfrac{1}{2}\bigl(\nu^\top S^{-1}\nu + \log\det(2\pi S)\bigr),
\end{equation}
computed in log-space to avoid underflow at $\log\mathcal L\sim -50$.

\paragraph{World-frame composition.}\label{sec:world_compose}
\file{gaussian\_state.py}\,$\to$\,\texttt{collapsed\_object\_world}
returns the canonical world-frame view consumed by every downstream
client (relations, slow tier, gravity-predict input, visualizer):
\begin{equation}\label{eq:world_compose}
\mu_{wo} = T_{wb}\,\mu_{bo}, \qquad
\Sigma_{wo} = \Ad(\mu_{bo}^{-1})\,\Sigma_{wb}\,\Ad(\mu_{bo}^{-1})^\top
              + \Sigma_{bo}.
\end{equation}
$\Sigma_{wb}$ is read from the current SLAM marginal at query time, so
the world cov tracks any SLAM drift / loop-closure shift even when no
new object observation has arrived. Combined with the
keyframe-mechanism freeze of $\Sigma_{bo}$ (\eqref{eq:proc}), an
unobserved static track's world cov grows \emph{only} via the
$\Sigma_{wb}$-lever — the user-intended behaviour.

\paragraph{World-frame overwrite (release hook).}
\file{gaussian\_state.py}\,$\to$\,\texttt{overwrite\_object\_pose(oid,
$T_{wo}$, $P_{wo}$)} forces the EKF state for one oid from a
world-frame $(T,P)$ input. Used by \file{orchestrator\_gaussian.py}
$\to$\,\texttt{\_maybe\_gravity\_predict} (and the RBPF analogue in
\file{orchestrator.py}) at the release transition to install
\eqref{eq:gravity_sigma}'s landing prior. The body-frame storage
becomes $\mu_{bo}=T_{wb}^{-1}T_{wo}$ and $\Sigma_{bo}=P_{wo}$ (the
right-trivialised tangent transport reduces to the identity at the
new $\mu_{bo}$). After the overwrite, the keyframe-mechanism freeze
in \eqref{eq:proc} keeps the gravity-supplied cov
($\sigma_{yaw}=\pi$, $\sigma_{xy}\!\sim\!5$–$10\,$cm) intact across
subsequent unobserved frames.

\paragraph{Per-track observation chain (diagnostic).}\label{sec:chain}
Each track also accumulates an append-only chain
$\mathcal{H}^{(i)} = [(\,k,\hat T^{co}_k,\hat R^{co}_k\,),\ldots]$ of
camera-frame ICP measurements (\file{pose\_update/obs\_chain.py}).
\textbf{The chain is NOT the canonical world-frame source} — that
role belongs to \texttt{collapsed\_object\_world}
(\eqref{eq:world_compose}). The chain is computed in parallel by the
visualizer (\file{tests/visualize\_ekf\_tracking.py}) for
cross-checking and is the natural substrate for a future
loop-closure-aware smoother (the slow-tier
\file{pose\_update/factor\_graph.py} consumes the same structure when
promoted).

\paragraph{Huber reweight.}
\begin{equation}\label{eq:huber_w}
w(d^2; G_{\text{in}}, G_{\text{out}}) = \begin{cases}
1 & d^2 \le G_{\text{in}}, \\
G_{\text{in}}/d^2 & G_{\text{in}} < d^2 \le G_{\text{out}}, \\
0 & d^2 > G_{\text{out}}.
\end{cases}
\end{equation}
Defaults: $G_{\text{in},t}=7.815$ ($\chi^2_3$ at 95\%),
$G_{\text{out},t}=21.108$.

\subsection*{II.6 Birth and prune}

Birth gates (\file{orchestrator.py}\,$\to$ \texttt{birth\_admissible}
and \file{birth\_gating.py}\,$\to$ \texttt{is\_near\_live\_track}):
\begin{enumerate}
\item Border: bbox not within \texttt{birth\_border\_margin\_px}=2~px of edge.
\item Confirm-$k$: tracker-side pending counter $\ge k = $
  \texttt{birth\_confirm\_k}=3 (counts the tracker's unmatched-streak
  frames, not perception's $n_{\text{obs}}$ which resets on SAM2 id reissue).
\item Score floor: rolling-max score $\ge $ \texttt{birth\_score\_min}=0.20.
\item Near-live: world-frame centroid $> $ \texttt{birth\_min\_dist\_m}=0.05~m
  from any same-label live track. For the held oid the radius widens to
  \texttt{held\_birth\_radius\_m}=0.25~m and $T_{we}$ is checked as a
  secondary anchor.
\item TTL: pending entries that go \texttt{birth\_pending\_ttl\_frames}=30
  frames without a matching arrival are evicted (one second at 30\,Hz;
  survives short perception dropouts).
\end{enumerate}
Survivors mint a fresh oid via \texttt{\_next\_track\_id()}, run ICP
once on admission (\file{icp\_pose.py}\,$\to$\,\texttt{PoseEstimate.estimate}
with \texttt{T\_co\_init=None}), and seed
\begin{equation}\label{eq:birth_r}
r_0 \;=\; \frac{\lambda_b\,\hat s_d}{\lambda_b\,\hat s_d + \lambda_c}
\end{equation}
(\file{bernoulli.py}\,$\to$\,\texttt{r\_birth}; defaults
$\lambda_b=\lambda_c=1$ → $r_0=\hat s_d$). Birth covariance defaults to
\begin{equation}\label{eq:birth_state}
P_{\text{new}} = \hat R^{co} \quad (\texttt{init\_cov\_from\_R=True}, \text{default})
\end{equation}
and falls back to $\mathrm{diag}((5\,\text{cm})^2{\times}3,(0.05\,\text{rad})^2{\times}3)$
when \texttt{init\_cov\_from\_R=False}.

\paragraph{Existence updates.}
For matched pairs (\eqref{eq:r_assoc};
\file{bernoulli.py}\,$\to$\,\texttt{r\_assoc\_update\_loglik})
\begin{equation}\label{eq:r_assoc}
r' = \frac{(1 - p_s) + p_s\,p_d\,\mathcal L\,r}{1 - p_s + p_s\,r},
\end{equation}
and for missed tracks (\eqref{eq:r_miss};
\file{bernoulli.py}\,$\to$\,\texttt{r\_miss\_update}; $\tilde p_d = p_d\,p_v$)
\begin{equation}\label{eq:r_miss}
r' = \frac{(1 - \tilde p_d)\,r}{1 - \tilde p_d\,r},
\end{equation}
with $p_v$ from \file{pose\_update/visibility.py} (see \eqref{eq:pv}).

\paragraph{Prune.}
$r_o < $ \texttt{r\_min}=$10^{-3}$ deletes, EXCEPT the held seed, whose
existence is floored at \texttt{r\_held\_floor}=0.5 (survives long
held-occlusion).

\subsection*{II.7 Self-merge}

\file{orchestrator.py}\,$\to$\,\texttt{\_self\_merge\_pass}. Inputs:
held seed $s$, protected pairs $\mathcal P$ derived from the relation
edge set $\mathcal E$ emitted by \file{relation\_orchestrator.py}
(\S II.8) — the parent/child convention is
\emph{parent = supported, child = supporter}, so a relation
$(p,c,\text{on})$ contributes the unordered pair
$(\min(p,c),\max(p,c))$ to $\mathcal P$:
\begin{equation*}
\mathcal{P} = \bigl\{\,(\min(p,c),\max(p,c))\;:\;(p,c,\cdot)\in\mathcal{E}\,\bigr\}.
\end{equation*}
Same-label pairs in $\mathcal P$ are protected from collapse — the scene
graph asserts they are physically distinct (apple-on-tray, etc.).
Information-sum fusion (with the keep mean as linearisation point) is
\begin{equation}\label{eq:self_merge_info}
\begin{aligned}
P^{-1}_{\text{new}} &= P^{-1}_{\text{keep}} + P^{-1}_{\text{drop}}, \\
\mu_{\text{new}} &= \mu_{\text{keep}}\,\Exp\!\bigl(P_{\text{new}}\,P^{-1}_{\text{drop}}\,\Log(\mu_{\text{keep}}^{-1}\mu_{\text{drop}})\bigr).
\end{aligned}
\end{equation}

\begin{algorithm}[H]
\caption{Self-merge pass.}
\begin{algorithmic}[1]
\For{each pair $(o_i, o_j)$ with $\ell_i = \ell_j$}
  \If{$(\min(o_i,o_j),\max(o_i,o_j))\in\mathcal{P}$} \textbf{continue} \EndIf
  \State $d \gets \|\mu_w(o_i) - \mu_w(o_j)\|_2$
  \If{$d > 0.05$} \textbf{continue} \EndIf
  \State Push $(d, d^2_t, o_i, o_j)$ as candidate.
\EndFor
\State Sort candidates by $d$ ascending.
\For{each $(d, d^2_t, o_i, o_j)$}
  \If{$o_i$ or $o_j$ already absorbed} \textbf{continue} \EndIf
  \State Choose keep$\gets s$ if $s\in\{o_i,o_j\}$, else higher-$r$, else lower oid.
  \State Information-sum-fuse $\mu, \Sigma$; merge label histograms / sam2\_tau / T\_oe / r.
  \State Append \{keep, drop, $d$, $d^2_t$\} to merges.
\EndFor
\end{algorithmic}
\end{algorithm}
The protected-pairs gate is what stops an apple-on-tray pair from
collapsing once their world centroids drift within 5~cm under
occlusion bias.

\subsection*{II.8 Visibility}\label{sec:vis}

\file{pose\_update/visibility.py}\,$\to$\,\texttt{visibility\_p\_v}.
For each track $i$, sample object-surface points
$\{x_j^{(i)}\}_{j=1}^M$ (paper specifies projecting the TSDF shape
summary $\mathcal{G}^{(i)}$; current implementation uses a Fibonacci
sphere of radius $r_{\text{obj}}$ around $\mu_{bo}$). Lift each through
$T_{bc}$ into camera, depth-test against the current depth image:
\begin{equation}\label{eq:pv}
p_v^{(i)} \;=\;
\frac{\#\bigl\{j : x_j^{(i)} \text{ in-frustum and passes depth test}\bigr\}}
     {\#\bigl\{j : x_j^{(i)} \text{ in-frustum}\bigr\}}
\quad\in\;[0,1].
\end{equation}
$p_v=1$ when no in-frustum sample is occluded (fully visible / no
evidence of occlusion); $p_v=0$ when fully out-of-FOV or fully occluded.
The continuous probability composes naturally with \eqref{eq:r_miss};
the four-state machine (present / occluded / absent / not-in-view)
collapses into a single weight on the miss-frame likelihood.

\subsection*{II.9 Relations}

\textbf{Trigger gate}
(\file{relation\_utils.py}\,$\to$\,\texttt{should\_recompute\_relations}):
fires at first call, release transitions, new-oid arrivals, or every
\texttt{relation\_every\_n\_frames}=90 frames. The grasp-transition
trigger was removed (LLM at grasp onset has incomplete information).

\textbf{Backend} (\file{relation\_orchestrator.py}\,+\,\file{relation\_client.py}).
On a triggered frame, the orchestrator builds normalised bboxes per
oid (from \texttt{det\_to\_oid}) and calls
\texttt{LLMRelationClient} or \texttt{RESTRelationClient}. The
backend returns $p\in[0,1]^{N\times N}$ with $p_{ij}$ = ``$j$ rests
on $i$''. Disk cache (\texttt{CachedRelationClient}) keyed by
$(\text{frame},N,\text{rounded bboxes})$.

\textbf{Convention swap}. Held-set expansion uses
parent=supported, child=supporter. The LLM's $(i,j)$ encodes the
opposite ($j$ rests on $i$). Edge emission swaps:
$$\mathrm{edge} = \mathrm{RelationEdge}(\text{parent}=j,\text{child}=i,\text{type}=\text{on},\text{score}=p_{ij}).$$
A regression here was responsible for held-set expansion failing —
the closure walks edges with $\text{child}\in S$, so a backward
emission means no propagation.

\textbf{EMA}
(\file{orchestrator.py}\,$\to$\,\texttt{RelationFilter}). For each
key $k=(\text{parent},\text{child},\text{type})$:
$$\mathrm{ema}_k(t) = \alpha\,\mathrm{raw}_k(t) + (1-\alpha)\,\mathrm{ema}_k(t-1),\quad \alpha=0.3.$$
Edges emitted when $\mathrm{ema}\ge 0.5$; pruned when
$\mathrm{ema}\le 0.01$. Half-life $\sim 2$ trigger calls
$\sim 180$ frames.

\textbf{Remap after merges}. When self-merge produces
$\{(\text{drop}_i,\text{keep}_i)\}$, every EMA key and emitted edge
has parent/child rewritten via the closure of \texttt{drop\_to\_keep}.
Self-loops drop, duplicate keys merge by max.

\subsection*{II.10 Held-set expansion}\label{sec:held}

\file{relation\_utils.py}\,$\to$\,\texttt{expand\_held\_with\_relations}.

\begin{algorithm}[H]
\begin{algorithmic}[1]
\State $S \gets \{s\}$
\For{$\textrm{iter}=1\ldots 8$}
  \State changed $\gets$ false
  \For{each edge $(p,c,t)\in\mathcal{E},\ t\in\{\text{in},\text{on}\}$}
    \If{$c\in S$ and $p\notin S$} $S \gets S\cup\{p\}$; changed$\gets$true \EndIf
  \EndFor
  \If{not changed} \textbf{break} \EndIf
\EndFor
\State \textbf{return} $S$
\end{algorithmic}
\end{algorithm}
The caller post-filters $S$ against currently-live oids to drop
references to pruned tracks the EMA hasn't yet decayed.

\subsection*{II.11 Gravity-aware predict at release}\label{sec:gravity}

When the gripper releases an object, the rigid-attachment predict no
longer applies and the static predict over-states the prior (the
object physically falls). This subsystem replaces the EKF mean and
covariance for the just-released oid with the post-settling
prediction in a single step at the
$\textsc{Releasing}\!\to\!\textsc{Idle}$ transition.

\paragraph{Voxel observability.}
\file{pose\_update/voxel\_observability.py} maintains a dense 3-state
grid (\textsc{Unseen}\,/\,\textsc{Empty}\,/\,\textsc{Occupied}) plus
per-voxel \texttt{n\_hit} + \texttt{n\_pass} counters with hysteresis
($\ge $ \texttt{n\_min\_hit}=2 confirming hits and $\ge $
\texttt{n\_min\_pass}=3 pass-through rays before transition). Updated
each frame from depth: each pixel's ray marks voxels traversed before
the depth hit as pass-pending and the voxel at the hit as
hit-pending. Default workspace $5\,\text{m}\times 5\,\text{m}\times
3\,\text{m}$ at $5\,\text{cm}$ voxel size.

\paragraph{Object dynamics property.}
\file{pose\_update/object\_dynamics.py} hosts a per-label table
mapping each detection label to $(e,\,\mu,\,\text{shape},\,r_{\text{obj}})$.
Restitution defaults from agricultural / engineering literature: apple
$e=0.30$, milkbox $0.40$, cola $0.50$, cup $0.40$, pot $0.40$,
flowerpot $0.30$. Unknown labels fall back to $e=0.40,\,r=0.05$
(\texttt{DEFAULT\_DYNAMICS}). A perception-side override hook lets a
learned estimator (e.g.\ Galileo) supersede the table per-detection.

\paragraph{Parametric dispersion model}
(\file{pose\_update/gravity\_predict.py}\,$\to$\,\texttt{predict\_landing\_pose}).
For drop height $h$ (resolved via voxel-driven raycast, below):
\begin{equation}\label{eq:gravity_sigma}
\begin{aligned}
v_{\text{impact}} &= \sqrt{2 g h}, \qquad
  t_{\text{fall}}  = \sqrt{2 h / g}, \\
\sigma_{\text{bounce}} &= \varepsilon_{\text{rough}}\,v_{\text{impact}} \big/ \max(0.1,\,1-e), \\
\sigma_{\text{shape}}  &= r_{\text{obj}}\,\text{footprint\_factor}(\text{shape}), \\
\sigma_{\text{rel-v}}  &= \tfrac{1}{2}\,\|\bm v_{\text{rel,xy}}\|\,t_{\text{fall}}, \\
\sigma_{xy}^2 &= \sigma_{\text{bounce}}^2 + \sigma_{\text{shape}}^2 + \sigma_{\text{rel-v}}^2, \\
\sigma_{z} &= \tfrac{1}{2}\,r_{\text{obj}}\,(1 - e/2), \\
\sigma_{\text{yaw}} &= \pi\,\bigl(1 - \exp\!\bigl(-h(1-e^2)/r_{\text{obj}}\bigr)\bigr).
\end{aligned}
\end{equation}
Default $\varepsilon_{\text{rough}} = 5\!\cdot\!10^{-3}\,\text{m}\,/\,(\text{m s}^{-1})$;
\texttt{shape\_footprint\_factor}: spherical $0.25$, cylindrical
$0.50$, box $0.70$, irregular $1.00$. Higher restitution increases
$\sigma_{\text{bounce}}$ — physically, a rubber ball bounces several
times each accruing scatter, while a clay ball lands once and stays.

\paragraph{Voxel-driven $h$ branch.}
The downward raycast (\texttt{VoxelObservability.raycast\_down})
classifies the column into one of four states; each adjusts $h$,
landing $z$, and the $\sigma_z$ floor:
\begin{itemize}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item \texttt{hit\_occupied}: clean surface at $h^\star$;
  $h = h^\star$, $\sigma_z$ from \eqref{eq:gravity_sigma}.
\item \texttt{mixed\_unseen}: surface known but UNSEEN voxels above
  it; $h = h^\star$, $\sigma_z$ extra-inflated by
  $(h_u - h^\star)/2$ to reflect path uncertainty.
\item \texttt{all\_unseen}: surface unknown;
  $h = (h_u + h_{\text{floor}})/2$ (\emph{midpoint}, not worst case),
  $\sigma_z = (h_u - h_{\text{floor}})/2$.
\item \texttt{all\_empty}: column observed empty all the way to the
  workspace floor; $h = h_{\text{floor}}$,
  $\sigma_z = r_{\text{obj}}$.
\end{itemize}
The midpoint choice for \texttt{all\_unseen} gives the EKF a useful
prior that subsequent observations can correct quickly, instead of
pushing the mean to an extreme that's unlikely to be right.

\paragraph{Trigger.}
Three production sites all detect the FSM exiting
$\{\textsc{Holding},\textsc{Releasing}\}$ and call into
\eqref{eq:gravity_sigma}:
\begin{itemize}[leftmargin=1.5em,itemsep=0pt,topsep=2pt]
\item \file{pose\_update/orchestrator.py}\,$\to$\,\texttt{\_maybe\_gravity\_predict} (RBPF backend) — replaces every particle's $(T_{wo}, P_{wo})$ via \texttt{RBPFState.overwrite\_object\_pose}.
\item \file{pose\_update/orchestrator\_gaussian.py}\,$\to$\,\texttt{\_maybe\_gravity\_predict} (Gaussian backend) — calls \texttt{GaussianState.overwrite\_object\_pose} (see §II.5 world-frame overwrite). Wired in \texttt{TwoTierOrchestratorGaussian.step} immediately after the fast tier.
\item \file{tests/visualize\_ekf\_tracking.py} (instrumented tracker harness) — mutates \texttt{state.objects[oid].\{mu\_bo, cov\_bo\}} inline.
\end{itemize}
Subsequent frames see the released track in
\texttt{unobserved\_oids}, so \eqref{eq:proc}'s keyframe freeze
preserves the gravity prior (in particular $\sigma_{yaw}=\pi$ does
\emph{not} pump $\sigma_{xy}$ via cross-coupling).

\paragraph{Toggle.}
\texttt{BernoulliConfig.gravity\_predict} (default \texttt{True})
gates the RBPF hook;
\texttt{TwoTierOrchestratorGaussian.gravity\_predict\_enabled}
(constructor flag, default \texttt{True}) gates the Gaussian hook.
\texttt{BernoulliConfig.workspace\_floor\_z} /
\texttt{TwoTierOrchestratorGaussian.workspace\_floor\_z} (default
$-1.0$\,m) sets the column terminator. With \texttt{voxel\_obs}
unset, both hooks short-circuit and the legacy static-Q-only release
behaviour is preserved exactly.

\subsection*{II.12 Gripper phase FSM}

\file{pose\_update/gripper\_state.py}\,$\to$\,\texttt{GripperPhaseTracker}.
Four phases:
$\textsc{Idle}\to\textsc{Grasping}\to\textsc{Holding}\to\textsc{Releasing}\to\textsc{Idle}$.
\textbf{Project assumption:} the gripper begins fully open; ANY closing
of the fingers registers as a grasp action. The state machine therefore
detects ``closed'' as a drop $\ge $ \texttt{close\_delta\_m}=$0.005$~m
below the rolling open baseline (5\,mm captures any non-trivial finger
motion while ignoring proprioception jitter), in addition to the
absolute thresholds. Driven by:
\begin{itemize}
\item Width threshold crossings with hysteresis
  (\texttt{closed\_width\_m}=$0.025$~m, \texttt{open\_width\_m}=$0.040$~m,
  \texttt{close\_delta\_m}=$0.005$~m below baseline).
\item Stability test (last \texttt{history\_size}=$5$ widths span
  $<$ \texttt{motion\_threshold\_m}=$0.01$~m).
\item Min-frames timer (\texttt{min\_transition\_frames}=$5$) holds the
  transition phases through brief instability.
\end{itemize}
At the \textsc{Idle}$\to$\textsc{Grasping} edge, the held oid is
selected by \texttt{GraspOwnerDetector}: rank detections by inside-
gripper-volume containment fraction (= count\_inside / total\_pts);
tiebreak by smallest mask area. Falls back to nearest-live-track
within $0.30$~m of $T_{we}$. Self-merges propagate via
\texttt{apply\_merges}: drop $=$ \texttt{held\_obj\_id} $\to$ switch
to the keep.

\subsection*{II.13 Frame-by-frame algorithm}

\begin{algorithm}[H]
\caption{One frame of the Bernoulli-EKF tracker (driver loop).}
\begin{algorithmic}[1]
\State \emph{Inputs}: $T_{wb}$, $T_{bc}$, $T_{bg}$, $w$, RGB, depth, detections.
\State $g \gets \textsc{GripperPhaseTracker}.\textsc{step}(w, \ldots)$
\State $\textsc{relation\_pipeline}.\textsc{maybe\_update}(\ldots)$
\State $\mathcal{H} \gets \textsc{Expand}(g.\text{held},\,\mathcal{E}_t)\cap\text{live oids}$
\State $\mathcal{U} \gets \{o : \texttt{frames\_since\_obs}[o] > 0\}$ \Comment{keyframe-frozen set}
\State \textsc{predict\_static}(skip\,$=\!\mathcal{H}$,\;unobserved\,$=\!\mathcal{U}$);\,\textsc{rigid\_attachment\_predict}(each $o\in\mathcal{H}$)
\State \textsc{suppress\_subpart\_detections};\,\textsc{centroid\_cam\_from\_mask};\,gather \texttt{dets\_for\_assoc}
\State \textsc{hungarian\_associate} $\to$ matches, unmatched\_tracks, unmatched\_dets
\For{each match $(o, d)$}
  \If{$o = g.\text{held}$ and (held-prefilter or innov-clamp triggers)} skip update \EndIf
  \State \textsc{update\_observation\_centroid}; update $r_o$ via match log-likelihood
\EndFor
\For{each unmatched track $o$}
  \State Update $r_o$ via miss equation; floor at \texttt{r\_held\_floor} if $o = g.\text{held}$
\EndFor
\For{each unmatched detection $d$}
  \If{\textsc{birth\_admissible}$(d)$ and not \textsc{is\_near\_live\_track}$(d)$}
    \State Mint new oid, run ICP once, set $r_0=0.4$
  \EndIf
\EndFor
\State Prune oids with $r_o < r_{\min}$ (skip held seed)
\State $\mathcal{P} \gets \{(\min(p,c),\max(p,c))\,:\,(p,c,\cdot)\in\mathcal{E}_t\}$
\State \textsc{\_self\_merge\_pass}$(\text{held}=g.\text{held},\,\text{protected}=\mathcal{P})$
\State $g$.\textsc{apply\_merges}(merges); \textsc{relation\_pipeline}.\textsc{remap\_after\_merges}(merges)
\State \textsc{\_maybe\_gravity\_predict}$(g)$ \Comment{release transition: overwrite the just-released oid via \eqref{eq:gravity_sigma}}
\end{algorithmic}
\end{algorithm}
